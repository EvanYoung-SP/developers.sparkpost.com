var sparky = new SparkPost("API KEY");
sparky.transmission.send({
  from: {name: "Sparky", email: "sparky@sparkpostbox.com"},
  subject: "Emailing like a champ.",
  html:"<html><body><p>Emailing awesomeness everywhere.</p></body></html>",
  recipients: [
    {name: "Your Name", email: "you@example.com"}
  ]
});
from sparkpost import SparkPost

sp = SparkPost('API KEY')

response = sp.transmissions.send(
  recipients=['someone@somedomain.com'],
  html='<html><body><p>Emailing awesomeness everywhere.</p></body></html>',
  from_email='test@sparkpostbox.com',
  subject='Emailing like a champ.'
)
use SparkPost\SparkPost;
use GuzzleHttp\Client;
use Ivory\HttpAdapter\Guzzle6HttpAdapter;

$httpAdapter = new Guzzle6HttpAdapter(new Client());
$sparky = new SparkPost($httpAdapter, ['key'=>'YOUR API KEY']);

$results = $sparky->transmission->send([
'from'=>'From Envelope <from@sparkpostbox.com>',

    [

      ]
    ]
  ]
]);
